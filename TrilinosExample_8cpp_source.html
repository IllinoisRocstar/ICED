528  double powerMethod(const Epetra_Operator& A, std::ostream& out, const int niters, const double tolerance) {
529  // An Operator doesn't have a Comm, but its domain Map does.
530  const Epetra_Comm& comm = A.OperatorDomainMap().Comm();
531  const int myRank = comm.MyPID();
532 
533  // Create three vectors for iterating the power method. Since the
534  // power method computes z = A*q, q should be in the domain of A and
535  // z should be in the range. (Obviously the power method requires
536  // that the domain and the range are equal, but it's a good idea to
537  // get into the habit of thinking whether a particular vector
538  // "belongs" in the domain or range of the matrix.) The residual
539  // vector "resid" is of course in the range of A.
540  Epetra_Vector q (A.OperatorDomainMap());
541  Epetra_Vector z (A.OperatorRangeMap());
542  Epetra_Vector resid (A.OperatorRangeMap());
543 
544  // Local error code for use below.
545  int lclerr = 0;
546 
547  // Fill the iteration vector z with random numbers to start. Don't
548  // have grand expectations about the quality of our pseudorandom
549  // number generator; this is usually good enough for eigensolvers.
550  //
551  // If this call fails, just let the code below finish before trying
552  // to catch the error globally. Ditto for other calls below.
553  lclerr = z.Random ();
554 
555  // lambda: the current approximation of the eigenvalue of maximum magnitude.
556  // normz: the 2-norm of the current iteration vector z.
557  // residual: the 2-norm of the current residual vector "resid"
558  double lambda = 0.0;
559  double normz = 0.0;
560  double residual = 0.0;
561 
562  const double zero = 0.0;
563  const double one = 1.0;
564 
565  // How often to report progress in the power method. Reporting
566  // progress requires computing a residual which can be expensive.
567  // However, if you don't compute the residual often enough, you
568  // might keep iterating even after you've converged.
569  const int reportFrequency = 10;
570 
571  // Do the power method, until the method has converged or the
572  // maximum iteration count has been reached.
573  for (int iter = 0; iter < niters; ++iter) {
574  // If you feel confident that your code is correct, you may omit
575  // everything having to do with lclerr, and just write the following:
576  //
577  // z.Norm2 (&normz); // Compute the 2-norm of z
578  // q.Scale (one / normz, z); // q := z / normz
579  // A.Apply (q, z); // z := A * q
580  // q.Dot (z, &lambda); // Approx. max eigenvalue
581 
582  lclerr = (lclerr == 0) ? z.Norm2 (&normz) : lclerr;
583  lclerr = (lclerr == 0) ? q.Scale (one / normz, z) : lclerr;
584  lclerr = (lclerr == 0) ? A.Apply (q, z) : lclerr;
585  lclerr = (lclerr == 0) ? q.Dot (z, &lambda) : lclerr;
586

588  // iterations, or if we've reached the maximum iteration count.
589  if (iter % reportFrequency == 0 || iter + 1 == niters) {

591  // everything having to do with lclerr, and just write the
592  // following:
593  //
594  // resid.Update (one, z, -lambda, q, zero); // z := A*q - lambda*q
595  // resid.Norm2 (&residual); // 2-norm of the residual vector
596 
597  lclerr = (lclerr == 0) ? resid.Update (one, z, -lambda, q, zero) : lclerr;
598  lclerr = (lclerr == 0) ? resid.Norm2 (&residual) : lclerr;
599 
600  if (myRank == 0) {
601  out << "Iteration " << iter << ":" << std::endl
602  << "- lambda = " << lambda << std::endl
603  << "- ||A*q - lambda*q||_2 = " << residual << std::endl;
604  }
605  }
606  if (residual < tolerance) {
607  if (myRank == 0) {
608  out << "Converged after " << iter << " iterations" << std::endl;
609  }
610  break;
611  } else if (iter + 1 == niters) {
612  if (myRank == 0) {
613  out << "Failed to converge after " << niters << " iterations" << std::endl;
614  }
615  break;
616  }
617  }

630  void TrilinosRedistributeExample(const Epetra_Comm& comm)
631  {
632  // The global number of rows in the matrix A to create. We scale
633  // this relative to the number of (MPI) processes, so that no matter
634  // how many MPI processes you run, every process will have 10 rows.
635  const global_ordinal_type numGblElts = 10 * comm.NumProc ();
636  // The global min global index in all the Maps here.
637  const global_ordinal_type indexBase = 0;
638 
639  // Local error code for use below.
640  //
641  // In the ideal case, we would use this to emulate behavior like
642  // that of Haskell's Maybe in the context of MPI. That is, if one
643  // process experiences an error, we don't want to abort early and
644  // cause the other processes to deadlock on MPI communication
645  // operators. Rather, we want to chain along the local error state,
646  // until we reach a point where it's natural to pass along that
647  // state with other processes. For example, if one is doing an
648  // MPI_Allreduce anyway, it makes sense to pass along one more bit
649  // of information: whether the calling process is in a local error
650  // state. Epetra's interface doesn't let one chain the local error
651  // state in this way, so we use extra collectives below to propagate
652  // that state. The code below uses very conservative error checks;
653  // typical user code would not need to be so conservative and could
654  // therefore avoid all the all-reduces.
655  int lclerr = 0;
656 
657  // Construct a Map that is global (not locally replicated), but puts
658  // all the equations on MPI Proc 0.
659  const int procZeroMapNumLclElts = (comm.MyPID () == 0) ?
660  numGblElts :
661  static_cast<global_ordinal_type> (0);
662  Epetra_Map procZeroMap (numGblElts, procZeroMapNumLclElts, indexBase, comm);
663 
664  // Construct a Map that puts approximately the same number of
665  // equations on each processor.
666  Epetra_Map globalMap (numGblElts, indexBase, comm);
667 
668  // Create a sparse matrix using procZeroMap.
669  Epetra_CrsMatrix* A = TrilinosCreateCrsMatrix (procZeroMap);
670  if (A == NULL) {
671  lclerr = 1;
672  }
673 
674  // Make sure that sparse matrix creation succeeded. Normally you
675  // don't have to check this; we are being extra conservative because
676  // this example is also a test. Even though the matrix's rows live
677  // entirely on Process 0, the matrix is nonnull on all processes in
678  // its Map's communicator.
679  int gblerr = 0;
680  (void) comm.MaxAll (&lclerr, &gblerr, 1);
681  if (gblerr != 0) {
682  throw std::runtime_error ("createCrsMatrix returned NULL on at least one "
683  "process.");
684  }
685 
686  //
687  // We've created a sparse matrix whose rows live entirely on MPI
688  // Process 0. Now we want to distribute it over all the processes.
689  //
690 
691  // Redistribute the matrix. Since both the source and target Maps
692  // are one-to-one, we could use either an Import or an Export. If
693  // only the source Map were one-to-one, we would have to use an
694  // Import; if only the target Map were one-to-one, we would have to
695  // use an Export. We do not allow redistribution using Import or
696  // Export if neither source nor target Map is one-to-one.
697 
698  // Make an export object with procZeroMap as the source Map, and
699  // globalMap as the target Map. The Export type has the same
700  // template parameters as a Map. Note that Export does not depend
701  // on the Scalar template parameter of the objects it
702  // redistributes. You can reuse the same Export for different
703  // Tpetra object types, or for Tpetra objects of the same type but
704  // different Scalar template parameters (e.g., Scalar=float or
705  // Scalar=double).
706  Epetra_Export exporter (procZeroMap, globalMap);
707 
708  // Make a new sparse matrix whose row map is the global Map.
709  Epetra_CrsMatrix B (Copy, globalMap, 0);
710 
711  // Redistribute the data, NOT in place, from matrix A (which lives
712  // entirely on Proc 0) to matrix B (which is distributed evenly over
713  // the processes).
714  //
715  // Export() has collective semantics, so we must always call it on
716  // all processes collectively. This is why we don't select on
717  // lclerr, as we do for the local operations above.
718  lclerr = B.Export (*A, exporter, Insert);
719 
720  // Make sure that the Export succeeded. Normally you don't have to
721  // check this; we are being extra conservative because this example
722  // example is also a test. We test both min and max, since lclerr
723  // may be negative, zero, or positive.
724  gblerr = 0;
725  (void) comm.MinAll (&lclerr, &gblerr, 1);
726  if (gblerr != 0) {
727  throw std::runtime_error ("Export() failed on at least one process.");
728  }
729  (void) comm.MaxAll (&lclerr, &gblerr, 1);
730  if (gblerr != 0) {
731  throw std::runtime_error ("Export() failed on at least one process.");
732  }
733 
734  // FillComplete has collective semantics, so we must always call it
735  // on all processes collectively. This is why we don't select on
736  // lclerr, as we do for the local operations above.
737  lclerr = B.FillComplete ();
738 
739  // Make sure that FillComplete succeeded. Normally you don't have
740  // to check this; we are being extra conservative because this
741  // example is also a test. We test both min and max, since lclerr
742  // may be negative, zero, or positive.
743  gblerr = 0;
744  (void) comm.MinAll (&lclerr, &gblerr, 1);
745  if (gblerr != 0) {
746  throw std::runtime_error ("B.FillComplete() failed on at least one process.");
747  }
748  (void) comm.MaxAll (&lclerr, &gblerr, 1);
749  if (gblerr != 0) {
750  throw std::runtime_error ("B.FillComplete() failed on at least one process.");
751  }
752 
753  if (A != NULL) {
754  delete A;
755  }
756  };

758  Epetra_CrsMatrix* TrilinosCreateCrsMatrix(const Epetra_Map& map)
759  {
760  const Epetra_Comm& comm = map.Comm ();
761 
762  // Create an Epetra_CrsMatrix using the Map, with dynamic allocation.
763  Epetra_CrsMatrix* A = new Epetra_CrsMatrix (Copy, map, 3);
764 
765  // The list of global indices owned by this MPI process.
766 
767  const global_ordinal_type* myGblElts = NULL;
768  global_ordinal_type numGblElts = 0;
769  #ifdef EXAMPLE_USES_64BIT_GLOBAL_INDICES
770  myGblElts = map.MyGlobalElements64 ();
771  numGblElts = map.NumGlobalElements64 ();
772  #else
773  myGblElts = map.MyGlobalElements ();
774  numGblElts = map.NumGlobalElements ();
775  #endif // EXAMPLE_USES_64BIT_GLOBAL_INDICES
776 
777  // The number of global indices owned by this MPI process.
778  const int numMyElts = map.NumMyElements ();
779 
780  // In general, tests like this really should synchronize across all
781  // processes. However, the likely cause for this case is a
782  // misconfiguration of Epetra, so we expect it to happen on all
783  // processes, if it happens at all.
784  if (numMyElts > 0 && myGblElts == NULL) {
785  throw std::logic_error ("Failed to get the list of global indices");
786  }
787 
788  // Local error code for use below.
789  int lclerr = 0;
790 
791  // Fill the sparse matrix, one row at a time.
792  double tempVals[3];
793  global_ordinal_type tempGblInds[3];
794  for (int i = 0; i < numMyElts; ++i) {
795  // A(0, 0:1) = [2, -1]
796  if (myGblElts[i] == 0) {
797  tempVals[0] = 2.0;
798  tempVals[1] = -1.0;
799  tempGblInds[0] = myGblElts[i];
800  tempGblInds[1] = myGblElts[i] + 1;
801  if (lclerr == 0) {
802  lclerr = A->InsertGlobalValues (myGblElts[i], 2, tempVals, tempGblInds);
803  }
804  if (lclerr != 0) {
805  break;
806  }
807  }
808  // A(N-1, N-2:N-1) = [-1, 2]
809  else if (myGblElts[i] == numGblElts - 1) {
810  tempVals[0] = -1.0;
811  tempVals[1] = 2.0;
812  tempGblInds[0] = myGblElts[i] - 1;
813  tempGblInds[1] = myGblElts[i];
814  if (lclerr == 0) {
815  lclerr = A->InsertGlobalValues (myGblElts[i], 2, tempVals, tempGblInds);
816  }
817  if (lclerr != 0) {
818  break;
819  }
820  }
821  // A(i, i-1:i+1) = [-1, 2, -1]
822  else {
823  tempVals[0] = -1.0;
824  tempVals[1] = 2.0;
825  tempVals[2] = -1.0;
826  tempGblInds[0] = myGblElts[i] - 1;
827  tempGblInds[1] = myGblElts[i];
828  tempGblInds[2] = myGblElts[i] + 1;
829  if (lclerr == 0) {
830  lclerr = A->InsertGlobalValues (myGblElts[i], 3, tempVals, tempGblInds);
831  }
832  if (lclerr != 0) {
833  break;
834  }
835  }
836  }
837 
838  // If any process failed to insert at least one entry, throw.
839  int gblerr = 0;
840  (void) comm.MaxAll (&lclerr, &gblerr, 1);
841  if (gblerr != 0) {
842  if (A != NULL) {
843  delete A;
844  }
845  throw std::runtime_error ("Some process failed to insert an entry.");
846  }
847 
848  // Tell the sparse matrix that we are done adding entries to it.
849  gblerr = A->FillComplete ();
850  if (gblerr != 0) {
851  if (A != NULL) {
852  delete A;
853  }
854  std::ostringstream os;
855  os << "A->FillComplete() failed with error code " << gblerr << ".";
856  throw std::runtime_error (os.str ());
857  }
858 
859  return A;
860  };
